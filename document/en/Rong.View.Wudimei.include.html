include file

grammar of include file:
{include file="/inc/top.html"}
it use for including a template.
file attribute contains the relative template path.it 's in the folder $wudimei->viewsDirectory. the application will read the template $wudimei->viewsDirectory.$file ;eg :$wudimei->viewsDirectory."/inc/top.html

Filename: /test/tpl_wudimei/include.php
<?php
/**
 * file encoding utf-8
 * 文件字符编码utf-8
 */
$PathToRongFramework = dirname(__FILE__)."/../../lib";

set_include_path(  "." . PATH_SEPARATOR . $PathToRongFramework .  PATH_SEPARATOR . get_include_path() );


require_once 'Rong/View/Wudimei.php';
$wudimei = new Rong_View_Wudimei();
$wudimei->compileDir = dirname(__FILE__) . "/templates/compiled";
$wudimei->viewsDirectory = dirname(__FILE__) .  "/templates";

$wudimei->leftDelimiter = "{";
$wudimei->rightDelimiter = "}";

//assing var msg to template
$wudimei->assign("site_name", "wudimei shop");
$wudimei->assign("welcome", "hello,welcome to wudimei shop!");


$wudimei->display("hello/include_sample.html");



Filename: /test/tpl_wudimei/templates/inc/top.html
<h1>{$site_name}</h1>


Filename: /test/tpl_wudimei/templates/hello/include_sample.html
<html>
	<head>
		<title>include sample</title>
	</head>
	<body>
		{include file="inc/top.html"}
		<div>
			{$welcome}
		</div>
	</body>
</html>
